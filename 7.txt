def has_digits(предложение):
    для i в диапазоне (0, 10):
        если str(i) в предложении:
            return f'текст с цифрой {i}'
    return 'текст без цифр'

def count(предложение):
    для i в диапазоне (0, 10):
        если str(i) в предложении:
            возвращение 1
    возврат 0

summ = 0
n = int(input("Введите количество предложений: "))
для i в диапазоне(1, n + 1):
    o = input(f'Введите предложение №{i} ')
    предложение = o
    print(has_digits(предложение))
    summ += count(предложение);
print(f'Количество предложений с цифрой {summ}')
